-- MÓDULO 1: LA TIENDA DE ABARROTES (CON CONTROL DE INVENTARIO)
CREATE TABLE tienda_productos (
    id_producto BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nombre TEXT NOT NULL,
    precio_venta NUMERIC(10, 2) NOT NULL,
    stock_actual INT DEFAULT 0,              -- Nuevo: Cantidad física en tienda

CREATE TABLE tienda_ventas (
    id_venta BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    total NUMERIC(10, 2) NOT NULL,
    fecha_venta TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE TABLE tienda_detalle_ventas (
    id_detalle BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    id_venta BIGINT REFERENCES tienda_ventas(id_venta) ON DELETE CASCADE,
    id_producto BIGINT REFERENCES tienda_productos(id_producto) ON DELETE SET NULL,
    cantidad INT NOT NULL,
    subtotal NUMERIC(10, 2) NOT NULL
);

-- MÓDULO 2: LA REPARADORA DE CALZADO Y TALABARTERÍA
CREATE TABLE taller_tickets (
    id_servicio BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nombre_cliente TEXT NOT NULL,
    telefono TEXT NULL,
    email TEXT NULL,                      
    producto TEXT NOT NULL,               
    servicio_solicitado TEXT NOT NULL,    
    observaciones TEXT NULL,              
    estado TEXT DEFAULT 'Recibido' CHECK (estado IN ('Recibido', 'En Proceso', 'Terminado', 'Entregado', 'Abandonado')),
    costo_mano_obra NUMERIC(10, 2) DEFAULT 0.00,
    costo_materiales NUMERIC(10, 2) DEFAULT 0.00,
    costo_total NUMERIC(10, 2) GENERATED ALWAYS AS (costo_mano_obra + costo_materiales) STORED,
    anticipo NUMERIC(10, 2) DEFAULT 0.00,
    fecha_recepcion TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    fecha_entrega DATE NULL,              
    fecha_terminado TIMESTAMP WITH TIME ZONE NULL 
);

-- MÓDULO 3: CONTROL DE GASTOS SEPARADOS
CREATE TABLE gastos_operativos (
    id_gasto BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    concepto TEXT NOT NULL,
    monto NUMERIC(10, 2) NOT NULL,
    tipo_departamento TEXT NOT NULL CHECK (tipo_departamento IN ('Tienda', 'Taller')),
    fecha_gasto TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);
